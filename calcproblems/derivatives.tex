
	\begin{problem}		\label{problem:derivative-of-logistic-function}
		Calculate the derivative of the functions $f(x) = \frac{1}{1+e^{-x}}$
		and $g(x) \eqdef f(a+bx)$. % = \frac{1}{1+e^{-(a+bx)}}$.
		% cf. https://math.stackexchange.com/questions/78575/derivative-of-sigmoid-function-sigma-x-frac11e-x
		%	\begin{hint}
		%	\end{hint}
		\begin{answer}$f'(x) = f(x)(1-f(x))$ and $g'(x) = g(x)(1-g(x))b$.\end{answer}
		%		\begin{solution}
		%		\end{solution}
	\end{problem}
